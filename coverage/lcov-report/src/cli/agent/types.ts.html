/**
 * Types for the OpenCode-powered installation agent
 */
 
export interface CodeContext {
  /** Sample of existing pages/components for style matching */
  sampleFiles: { path: string; content: string }[];
  /** Import style detected (relative vs alias) */
  importStyle: "relative" | "alias";
  /** Path alias if detected (e.g., @/, ~/) */
  pathAlias?: string;
  /** Component naming convention */
  componentNaming: "PascalCase" | "kebab-case" | "camelCase";
  /** File naming convention */
  fileNaming: "PascalCase" | "kebab-case" | "camelCase";
  /** Export style */
  exportStyle: "default" | "named" | "mixed";
  /** Existing layout/wrapper components */
  layouts: string[];
  /** API route patterns */
  apiPatterns: { path: string; content: string }[];
  /** Quote style used in imports */
  quoteStyle: "single" | "double";
  /** Whether files use semicolons */
  semicolons: boolean;
  /** Indentation style */
  indentation: "2spaces" | "4spaces" | "tabs";
}
 
export interface DesignSystemInfo {
  /** Detected UI library: shadcn, chakra, mantine, mui, radix, headlessui, daisyui, or null */
  library: string | null;
  /** Known UI components found in the project (e.g., Button, Card, Badge) */
  components: string[];
  /** Whether a Header/Navbar component exists */
  hasHeader: boolean;
  /** Whether a Footer component exists */
  hasFooter: boolean;
  /** Whether a Navigation component exists */
  hasNavigation: boolean;
  /** Path to the main navigation file, if detected */
  navigationFile: string | null;
}
 
export interface I18nInfo {
  /** i18n library detected: next-intl, next-i18next, nuxt-i18n, svelte-i18n, paraglide, custom, or null */
  library: string | null;
  /** Detected locale codes */
  locales: string[];
  /** Whether routes are prefixed with a locale segment (e.g., /fr/blog) */
  routePrefix: boolean;
}
 
export interface MonorepoInfo {
  /** Whether the project is inside a monorepo */
  isMonorepo: boolean;
  /** Monorepo tool: turborepo, nx, lerna, pnpm-workspaces, or null */
  tool: string | null;
}
 
export interface RouteStructure {
  /** Discovered route groups in the app dir, e.g. ["(marketing)", "(app)"] */
  routeGroups: string[];
  /** The route group where most pages live, e.g. "(marketing)" or null */
  primaryRouteGroup: string | null;
  /**
   * Computed base path for creating blog pages.
   * Combines srcPrefix + app/ + routeGroup + localePrefix.
   * E.g. "src/app/(marketing)/[locale]/" or "src/app/" or "app/"
   */
  appDirPrefix: string;
  /**
   * Computed base path for creating API routes.
   * E.g. "src/app/api/" or "app/api/"
   */
  apiDirPrefix: string;
  /** The exact layout file that should wrap BlogProvider (closest to blog route segment) */
  layoutFile: string | null;
  /** Whether middleware.ts/.js exists at project root */
  hasMiddleware: boolean;
  /** Catch-all routes that might conflict with /blog, e.g. ["app/[...slug]", "src/app/[[...slug]]"] */
  catchAllRoutes: string[];
  /** Default dev server port for the detected framework */
  devPort: number;
  /** Public env var prefix for the detected framework (e.g. NEXT_PUBLIC_, PUBLIC_, NUXT_PUBLIC_) */
  envPrefix: string;
}
 
export interface ProjectInfo {
  /** Detected framework: nextjs, remix, astro, nuxt, sveltekit */
  framework: string | null;
  /** Router type for Next.js: pages or app */
  router?: string;
  /** Styling solution: tailwind, css-modules, styled-components, etc. */
  styling: string | null;
  /** Whether TypeScript is used */
  typescript: boolean;
  /** Package manager: npm, yarn, pnpm, bun */
  packageManager: string;
  /** Existing dependencies */
  dependencies: Record<string, string>;
  /** Project structure info */
  structure: {
    hasAppDir: boolean;
    hasPagesDir: boolean;
    hasSrcDir: boolean;
    configFiles: string[];
  };
  /** Existing blog-related files */
  existingBlogFiles: string[];
  /** Deep code analysis context */
  codeContext?: CodeContext;
  /** Project name from package.json */
  projectName?: string;
  /** Design system / UI library detected in the project */
  designSystem: DesignSystemInfo;
  /** Dominant data-fetching pattern */
  dataFetching: "swr" | "react-query" | "trpc" | "apollo" | "native";
  /** Internationalization configuration, if detected */
  i18n: I18nInfo | null;
  /** Monorepo detection */
  monorepo: MonorepoInfo | null;
  /** Tailwind CSS major version (3, 4, or null if not installed) */
  tailwindVersion: 3 | 4 | null;
  /** Theme / dark-mode strategy */
  themeMode: "light" | "dark" | "system" | "unknown";
  /** Computed route structure — resolved paths, route groups, catch-all detection */
  routeStructure: RouteStructure;
}
